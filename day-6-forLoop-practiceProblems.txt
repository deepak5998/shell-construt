#Program_1
#Print table of the powers of 2

#!/bin/bash 
read -p "Enter the number: " n
for (( i=1; i<=$n; i++ ))
do
res=$(echo |awk '{ print 2^i }' i=$i)
echo "2^$i=$res"
done
printf "\n"

#!/bin/bash

for (( i=1; i<=$n; i++))

fi
done

#Program_4
#Prime nos. in a given range

#!/bin/bash 

read -p "Enter first limit: " x
read -p "Enter last limit: " y
for (( i=$x; i<=$y; i++ ))
do
count=0;
	for (( j=2; j<=$i/2; j++))
	do
	rem=$(($i%$j))
		if [ $rem -eq 0 ]
		then
			count=$(($count+1))
		fi
	done
if [ $count -eq 0 ]
then
	echo "$i is Prime"
fi
done
printf "\n"
